% Define some often used physical quantities

\NewDocumentCommand\Ham{}{\mathbf{H}}
\NewDocumentCommand\SO{}{\mathbf{S}}
\NewDocumentCommand\DM{}{\boldsymbol{\rho}}
\NewDocumentCommand\EDM{}{\boldsymbol{\mathcal{E}}}
\NewDocumentCommand\elec{}{\mathfrak{e}}

\DeclareMathOperator{\Var}{Var}
\DeclareMathOperator{\Tr}{Tr}
\DeclareMathOperator*{\oppropto}{\propto}
\DeclareMathOperator\Res{R}
\DeclareMathOperator\RRes{\Delta R}

% Declare mathematical operators
\NewDocumentCommand\bra{m}{\langle#1|}
\NewDocumentCommand\ket{m}{|#1\rangle}
\NewDocumentCommand\kb{mm}{\ket{#1}\bra{#2}}
\NewDocumentCommand\bk{mm}{\langle#1|#2\rangle}
